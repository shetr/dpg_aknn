61template<typename FloatT, int Dim, typename ObjData = Empty>
62DistObj<FloatT, Dim, ObjData> LinearFindNearestNeighborInRangeWithDist(const PointObj<FloatT, Dim, ObjData>* objsBeg, const PointObj<FloatT, Dim, ObjData>* objsEnd, const Vec<FloatT, Dim>& queryPoint)
63{
64 PointObj<FloatT, Dim, ObjData> nnObj = *objsBeg;
65 FloatT minDist = queryPoint.DistSquared(nnObj.point);
66 for (const PointObj<FloatT, Dim, ObjData>* objIt = objsBeg + 1; objIt != objsEnd; ++objIt) {
67 FloatT dist = queryPoint.DistSquared(objIt->point);
68 if (dist < minDist) {
69 minDist = dist;
70 nnObj = *objIt;
71 }
72 }
73 return { minDist, nnObj };
74}

113template<typename FloatT, int Dim, typename ObjData>
114void PushChildsToNodeQueue(const BBDTree<FloatT, Dim, ObjData>& tree, const Vec<FloatT, Dim>& queryPoint, const DistNode<FloatT, Dim>& distNode, DistNodePriQueue<FloatT, Dim>& nodeQueue)
115{
116 const Node* node = tree.GetNode(distNode.nodeIdx);
117 const InnerNode* innerNode = (const InnerNode*)node;
118 Box leftBox = distNode.box;
119 Box rightBox = distNode.box;
120 FloatT distLeft;
121 FloatT distRight;
122 if (node->GetType() == NodeType::SPLIT)
123 {
124 const SplitNode* splitNode = (const SplitNode*)node;
125 int splitDim = splitNode->GetSplitDim();
126 FloatT half = (distNode.box.min[splitDim] + distNode.box.max[splitDim]) / 2;
127 leftBox.max[splitDim] = half;
128 rightBox.min[splitDim] = half;
129
130 distLeft = leftBox.SquaredDistance(queryPoint);
131 distRight = rightBox.SquaredDistance(queryPoint);
132 }
133 else if (node->GetType() == NodeType::SHRINK)
134 {
135 const ShrinkNode<FloatT, Dim>* shrinkNode = (const ShrinkNode<FloatT, Dim>*)node;
136 leftBox = shrinkNode->GetShrinkBox();
137
138 distLeft = leftBox.SquaredDistance(queryPoint);
139 distRight = distNode.dist;
140 }
141 if (innerNode->HasLeftChild())
142 nodeQueue.push({distLeft, distNode.nodeIdx + GetNodeOffset<FloatT, Dim>(node->GetType()), leftBox});
143 if (innerNode->GetRightChildIndex() != 0)
144 nodeQueue.push({distRight, innerNode->GetRightChildIndex(), rightBox});
145}

148template<typename FloatT, int Dim, typename ObjData = Empty, bool measureStats = false>
149PointObj<FloatT, Dim, ObjData> FindAproximateNearestNeighbor(const BBDTree<FloatT, Dim, ObjData>& tree, const Vec<FloatT, Dim>& queryPoint, FloatT epsilon, TraversalStats<FloatT, Dim>& stats)
150{
151 PointObj<FloatT, Dim, ObjData> ann;
152 FloatT minDist = std::numeric_limits<FloatT>::infinity();
153 DistNodePriQueue<FloatT, Dim> nodeQueue;
154 DistNode<FloatT, Dim> rootNode{tree.GetBBox().SquaredDistance(queryPoint), 0, tree.GetBBox()};
155 nodeQueue.push(rootNode);
156 while (!nodeQueue.empty())
157 {
158 DistNode<FloatT, Dim> distNode = nodeQueue.top();
159 const Node* node = tree.GetNode(distNode.nodeIdx);
160 nodeQueue.pop();
161
162 if (measureStats)
163 {
164 ++stats.traversalSteps;
165 stats.visitedNodes.push_back(distNode.box);
166 }
167
168 if (distNode.dist > minDist / (1 + epsilon)) {
169 break;
170 }
171
172 if (node->GetType() == NodeType::LEAF)
173 {
174 const LeafNode* leafNode = (const LeafNode*)node;
175 DistObj<FloatT, Dim, ObjData> localNN = LinearFindNearestNeighborInRangeWithDist<FloatT, Dim, ObjData>(
176 tree.GetObj(leafNode->GetPointsBegIndex()), tree.GetObj(leafNode->GetPointsEndIndex()), queryPoint);
177 if (localNN.dist < minDist)
178 {
179 minDist = localNN.dist;
180 ann = localNN.obj;
181 }
182
183 if (measureStats)
184 ++stats.visitedLeafs;
185 }
186 else
187 {
188 PushChildsToNodeQueue(tree, queryPoint, distNode, nodeQueue);
189 }
190 }
191 return ann;
192}

210{
