152template<typename FloatT, int Dim>
153index_t GetNodeOffset(NodeType nodeType)
154{
155 static_assert(sizeof(SplitNode) == sizeof(Node));
156 static_assert(sizeof(InnerNode) == sizeof(Node));
157 static_assert(sizeof(ShrinkNode<FloatT, Dim>) % sizeof(Node) == 0);
158 static_assert(sizeof(LeafNode) == 2*sizeof(Node));
159 switch (nodeType)
160 {
161 case NodeType::SPLIT:
162 return sizeof(SplitNode) / sizeof(Node);
163 case NodeType::SHRINK:
164 return sizeof(ShrinkNode<FloatT, Dim>) / sizeof(Node);
165 case NodeType::LEAF:
166 return sizeof(LeafNode) / sizeof(Node);
167 default:
168 break;
169 }
170 return 0;
171}

176template<typename FloatT, int Dim, typename FuncT, typename ObjData = Empty>
177PointObj<FloatT, Dim, ObjData>* SplitPoints(PointObj<FloatT, Dim, ObjData>* beg, PointObj<FloatT, Dim, ObjData>* end, FuncT isLeft)
178{
179 int size = end - beg;
180 int left = 0;
181 int right = size - 1;
182 while (left <= right) {
183 bool putLeft = isLeft(beg[left].point);
184 if (putLeft) {
185 ++left;
186 } else {
187 std::swap(beg[left], beg[right--]);
188 }
189 }
190 return beg + left;
191}

356 index_t BuildBasicSplitTreeR(SplitState<FloatT, Dim, ObjData> state)
357 {
358 if (state.size() == 0) {
359 // indicate that there is no child with 0 index (only root has 0 index and it can't be child of any node)
360 return 0;
361 } else if (state.size() <= _leafMaxSize) {
362 // add leaf node if number of points is small enough
363 return AddLeafNode(state.pointsBeg - _objs.data(), state.pointsEnd - _objs.data());
364 } else {
365 BoxSplit<FloatT, Dim> split = state.box.Split();
366 index_t splitNodeIndex = AddSplitNode(split.dim);
367 // split points
368 PointObjT* splitTo = SplitPoints(state.pointsBeg, state.pointsEnd, [&split](const Vec<FloatT, Dim>& point) {
369 return point[split.dim] < split.value;
370 });
371 BuildBasicSplitChilds(splitNodeIndex, split.left, split.right, state.pointsBeg, state.pointsEnd, splitTo);
372 return splitNodeIndex;
373 }

394 void SetBiggerSplit(SplitState<FloatT, Dim, ObjData>& state, BoxSplit<FloatT, Dim>& split, PointObjT*& splitTo)
395 {
396 split = state.box.Split();
397 splitTo = SplitPoints(state.pointsBeg, state.pointsEnd, [&split](const Vec<FloatT, Dim>& point) {
398 return point[split.dim] < split.value;
399 });
400 index_t leftSize = splitTo - state.pointsBeg;
401 index_t rightSize = state.pointsEnd - splitTo;
402 bool leftBigger = leftSize >= rightSize;
403 if (leftBigger) {
404 state.box = split.left;
405 state.pointsEnd = splitTo;
406 } else {
407 state.box = split.right;
408 state.pointsBeg = splitTo;
409 }
410 }

413 index_t BuildMidpointSplitTreeR(SplitState<FloatT, Dim, ObjData> state)
414 {
415 if (state.size() == 0) {
416 // indicate that there is no child with 0 index (only root has 0 index and it can't be child of any node)
417 return 0;
418 } else if (state.size() <= _leafMaxSize) {
419 // add leaf node if number of points is small enough
420 return AddLeafNode(state.pointsBeg - _objs.data(), state.pointsEnd - _objs.data());
421 } else {
422 BoxSplit<FloatT, Dim> split;
423 PointObjT* splitTo;
424 SplitState<FloatT, Dim, ObjData> splitState = state;
425 int splitCount = 0;
426 // split until number of points is <= 2/3 total number of points in current node
427 while (3 * splitState.size() > 2 * state.size() && splitState.size() > _leafMaxSize) {
428 SetBiggerSplit(splitState, split, splitTo);
429 ++splitCount;
430 }
431 // if there was only 1 split, we can simply create split node
432 if (splitCount == 1) {
433 index_t splitNodeIndex = AddSplitNode(split.dim);
434 BuildMidpointSplitChilds(splitNodeIndex, split.left, split.right, state.pointsBeg, state.pointsEnd, splitTo);
435 return splitNodeIndex;
436 } else { // otherwise we have a shrink node
437 // split the points according to the final box (inside box is on left, outside on right)
438 PointObjT* insideBoxTo = SplitPoints(state.pointsBeg, state.pointsEnd, [&splitState](const Vec<FloatT, Dim>& point) {
439 return splitState.box.Includes(point);
440 });
441 index_t shrinkNodeIndex = AddShrinkNode(splitState.box);
442 BuildMidpointSplitChilds(shrinkNodeIndex, splitState.box, state.box, state.pointsBeg, state.pointsEnd, insideBoxTo);
443 return shrinkNodeIndex;
444 }
445 }

449 void GetStatsR(BBDTreeIntermediateStats& stats, index_t nodeIndex, int depth) const {
450 const Node* node = GetNode(nodeIndex);
451
452 if (node->GetType() == NodeType::LEAF)
453 {
454 const LeafNode* leafNode = (const LeafNode*)node;
455 stats.leafSizesSum += GetLeafSize(leafNode);
456 ++stats.leafNodeCount;
457 stats.depthSum += depth;
458 stats.maxDepth = std::max(stats.maxDepth, depth);
459 }
460 else
461 {
462 const InnerNode* innerNode = (const InnerNode*)node;
463 ++stats.innerNodeCount;
464 if (node->GetType() == NodeType::SPLIT) {
465 ++stats.splitNodeCount;
466 }
467 else if (node->GetType() == NodeType::SHRINK) {
468 ++stats.shrinkNodeCount;
469 }
470
471 if (innerNode->HasLeftChild())
472 GetStatsR(stats, nodeIndex + GetNodeOffset<FloatT, Dim>(node->GetType()), depth + 1);
473 else
474 ++stats.nullCount;
475
476 if (innerNode->GetRightChildIndex() != 0)
477 GetStatsR(stats, innerNode->GetRightChildIndex(), depth + 1);
478 else
479 ++stats.nullCount;
480 }
481 }
