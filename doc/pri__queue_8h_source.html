124 if (IsEmpty()) {
125 _heap.push_back(value);

128 int i = GetSize();
129 _heap.push_back(value);
130 if (_isLeftSmaller(value, _first)) {
131 _first = value;
132 } else {
133 // ensure the heap property that parent is larger
134 while (i > 0) {
135 int p = GetParent(i);
136 if (_isLeftSmaller(_heap[i], _heap[p]))
137 break;
138 else
139 std::swap(_heap[i], _heap[p]);
140 i = p;
141 }
142 }
143 } else if (_isLeftSmaller(value, _heap[0])) { // insert only if the value is smaller than largest element
144 _heap[0] = value;
145 // update first
146 if (_isLeftSmaller(value, _first)) {
147 _first = value;
148 }
149 // ensure the heap property that parent is larger than its childs
150 int i = 0;
151 for (int l = GetLeftChild(i); l < GetSize(); l = GetLeftChild(i)) {
152 int r = l + 1;
153 int larger = r < GetSize() ? (_isLeftSmaller(_heap[r], _heap[l]) ? l : r) : l;
154 if (_isLeftSmaller(_heap[larger], _heap[i]))
155 break;
156 else
157 std::swap(_heap[i], _heap[larger]);
158 i = larger;
159 }
160 }
161 }
162private:
163 int GetParent(int i) const { return (i - 1) >> 1; }
